Node.js Integration Example

const axios = require('axios');
const API_URL = 'https://your-domain.vercel.app/api';

async function validateBot(botId) {
    try {
        // 1. Check License Status
        const response = await axios.get(`${API_URL}/check/${botId}`);
        
        if (!response.data.status) {
            console.error('⛔ Access Denied: Bot is Blacklisted');
            return false;
        }

        console.log('✅ Access Granted:', response.data.message);
        return true;

    } catch (error) {
        console.error('Connection Error:', error.message);
        return false;
    }
}

// 2. Register New Bot (First Run)
async function registerBot() {
    await axios.post(`${API_URL}/register`, {
        botId: "12345",
        ownerId: "67890",
        storeName: "My Store",
        ownerUsername: "@admin"
    });
}
